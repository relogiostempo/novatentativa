    const langButtons = document.querySelectorAll('[data-lang-btn]');
    function setLang(lang){
      document.documentElement.lang = (lang === 'pt') ? 'pt-PT' : (lang==='es'? 'es-ES' : 'en-GB');
      // toggle all elements with data-lang attribute
      document.querySelectorAll('[data-lang]').forEach(el=>{
        // original text in HTML is PT by default
        // we will store PT in data-pt attribute to allow switching back
        const key = el.getAttribute('data-lang-key') || null;
        // if element has translations in attributes, prefer those
        const textAttr = el.getAttribute('data-text-'+lang);
        if(textAttr!==null){ el.textContent = textAttr; return; }
        // else fallback to small mapping for nav
        // (for larger sites you would externalize strings)
      });
      // update nav & title minimal
      document.querySelectorAll('a[href="#services"]').forEach(a=>a.textContent = translations[lang].nav.services);
      document.querySelectorAll('a[href="#portfolio"]').forEach(a=>a.textContent = translations[lang].nav.portfolio);
      document.querySelectorAll('a[href="#about"]').forEach(a=>a.textContent = translations[lang].nav.about);
      document.querySelectorAll('a[href="#contact"]').forEach(a=>a.textContent = translations[lang].nav.contact);
      document.title = translations[lang].title;
    }
    langButtons.forEach(b=>{
      b.addEventListener('click', ()=> setLang(b.getAttribute('data-lang-btn')));
    });
    // default language PT
    setLang('pt');

    // lightbox
    document.querySelectorAll('.thumb img').forEach(img=>{
      img.addEventListener('click', ()=>{
        const src = img.getAttribute('data-full') || img.src;
        const lb = document.getElementById('lightbox');
        lb.querySelector('img').src = src;
        lb.style.display = 'flex';
        lb.setAttribute('aria-hidden','false');
      });
    });
    document.getElementById('lightbox').addEventListener('click', (e)=>{
      if(e.target.tagName==='IMG') return;
      e.currentTarget.style.display = 'none';

    });
